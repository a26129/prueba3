CREATE TABLE CLIENTES(
DNI VARCHAR2(9) PRIMARY KEY,
NOMBRE VARCHAR2(25),
APELLIDOS VARCHAR2(25),
DIRECCION VARCHAR2(25),
EDAD NUMBER(3));

CREATE TABLA PEDIDO(
ID_PEDIDO VARCHAR2(9) PRIMARY KEY,
DNI VARCHAR2(9) REFERENCES CLIENTES (DNI),
ID_LINEA VARCHAR2(9),
PRECIO_TOTAL NUMBER(5));

CREATE TABLA PRODUCTOS(
ID_PRODUCTO VARCHAR2(9) PRIMARY KEY,
NOMBRE VARCHAR2(25),
PRECIO NUMBER(3));

CREATE TABLA LINEAS_PEDIDO(
ID_LINEA VARCHAR2(9) PRIMARY KEY,
ID_PEDIDO VARCHAR2(9) REFERENCES PEDIDO (ID_PEDIDO),
ID_PRODUCTO VARCHAR2(9) REFERENCES PRODUCTOS (ID_PRODUCTO),
CANTIDAD NUMBER(4));

ALTER TABLE PEDIDO
ID_LINEA REFERENCES LINEAS_PEDIDO (ID_LINEA);
